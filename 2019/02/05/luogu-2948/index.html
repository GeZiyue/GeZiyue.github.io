class Class {
public:
    int begin, end, abi;
    friend bool operator<(Class, Class);
}c[M];
// иҜҫзЁӢзұ»еһӢ, еҸҜд»Ҙи§ҶдҪңstruct
int dp[N][M];
int lastmin[M];
int main () {
    int n, m, t;
    read(t);
    read(n);
    read(m);
    for (int i=1; i<=n; i++) {
        int begin, last, abi;
        read(begin);
        read(last);
        read(abi);
        c[i].begin=begin;
        c[i].end=begin+last;
        c[i].abi=abi;
    }
    sort(c+1, c+n+1, [](Class i, Class j) {
        return i.end<j.end;
    });// з»ҷиҜҫзЁӢжҢүз»“жқҹж—¶й—ҙжҺ’еәҸ, з”ЁеҲ°дәҶеҢҝеҗҚ(lambda)еҮҪж•°, йңҖиҰҒејҖC++11
    for (int i=1; i<=100; i++) {
        lastmin[i]=inf;
    }
    for (int i=1; i<=m; i++) {
        int need, last;
        read(need);
        read(last);
        lastmin[need]=min(lastmin[need], last);
    }
    memset(dp, -1, sizeof(dp));
    for (int i=0; i<=t; i++) {
        dp[i][1]=0;
    }
    int ans=0;
    for (int i=1, j=1; i<=t; i++) {// jиЎЁзӨәдёӢдёҖдёӘиҜҫзЁӢзҡ„зј–еҸ·
        if (c[j].end==i) {
            for (int k=1; k<=100; k++) {
                dp[i][c[j].abi]=max(dp[i][c[j].abi], dp[c[j].begin][k]);
            }
            for (int k=1; k<c[j].abi; k++) {
                dp[i][k]=max(dp[i][k], dp[i][c[j].abi]);
            }
            j++;
        }// дёҠиҜҫ
        for (int k=1; k<=100; k++) {
            if (i-lastmin[k]>=0) {
                if (dp[i-lastmin[k]][k]!=-1) {
                    dp[i][k]=max(dp[i][k], dp[i-lastmin[k]][k]+1);
                    ans=max(ans, dp[i][k]);
                }
            }
        }// ж»‘йӣӘ
    }
    write(ans);
    EL;
    return 0;
}
